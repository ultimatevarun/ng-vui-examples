Overview

NgVuiGrid is a powerful, feature-rich Angular 20+ data grid component with advanced template support, action columns, sorting, filtering, pagination, virtual scrolling, and export capabilities. Built with standalone components and modern Angular architecture.

Basic Usage

HTML Template:

<!-- Basic Grid Implementation -->
<div class="container mx-auto p-6">
  <div class="mb-4">
    <h1 class="text-2xl font-bold text-gray-900">Employee Directory</h1>
    <p class="text-gray-600">Manage your employee information</p>
  </div>

  <!-- Basic Grid -->
  <ng-vui-grid
    [rowData]="data"
    [columns]="columnDefs"
    [options]="gridOptions"
    [loading]="loading"
    (gridReady)="onGridReady($event)"
    (cellClicked)="onCellClicked($event)"
    (cellValueChanged)="onCellValueChanged($event)"
    (rowSelected)="onRowSelected($event)"
    (exportRequested)="onExport($event)"
    class="shadow-lg rounded-lg">
  </ng-vui-grid>

  <!-- Selection Info -->
  <div class="mt-4 p-4 bg-blue-50 rounded-lg" *ngIf="selectedRows.length > 0">
    <p class="text-blue-800">
      {{ selectedRows.length }} row(s) selected
    </p>
    <div class="mt-2 space-x-2">
      <button class="btn btn-sm btn-primary" (click)="bulkEdit()">
        Edit Selected
      </button>
      <button class="btn btn-sm btn-danger" (click)="bulkDelete()">
        Delete Selected
      </button>
    </div>
  </div>
</div>

TypeScript Component:

@Component({
  selector: 'app-employee-grid',
  templateUrl: './employee-grid.component.html',
  styleUrls: ['./employee-grid.component.css']
})
export class EmployeeGridComponent {
  loading = false;
  selectedRows: any[] = [];

  data = [
    { id: 1, name: 'John Doe', age: 30, email: 'john@example.com', department: 'Engineering' },
    { id: 2, name: 'Jane Smith', age: 25, email: 'jane@example.com', department: 'Marketing' },
    { id: 3, name: 'Bob Johnson', age: 35, email: 'bob@example.com', department: 'Sales' }
  ];

  columnDefs: ColumnDefinition[] = [
    { field: 'name', headerName: 'Full Name', sortable: true, filterable: true },
    { field: 'age', headerName: 'Age', type: 'number', width: 80 },
    { field: 'email', headerName: 'Email', type: 'email', width: 200 },
    { field: 'department', headerName: 'Department', sortable: true, filterable: true }
  ];

  gridOptions: GridOptions = {
    pagination: true,
    paginationPageSize: 10,
    enableSorting: true,
    enableFiltering: true,
    enableSelection: true,
    header: {
      title: 'Employee Directory',
      showRefresh: true,
      showExport: true,
      showColumnSelector: true
    },
    footer: {
      showPagination: true,
      showRowCount: true,
      showPageSize: true,
      pageSizes: [5, 10, 25, 50]
    }
  };
}

Column Definition

The ColumnDefinition interface defines the structure and behavior of grid columns.

Basic Column Properties

interface ColumnDefinition {
  field: string;                    // Data field name (required)
  headerName?: string;              // Display name in header
  headerTooltip?: string;           // Tooltip for header
  visible?: boolean;                // Column visibility
  width?: number;                   // Column width in pixels
  minWidth?: number;                // Minimum width
  maxWidth?: number;                // Maximum width
  resizable?: boolean;              // Allow column resizing
  sortable?: boolean;               // Enable sorting
  filterable?: boolean;             // Enable filtering
  editable?: boolean | Function;    // Enable cell editing
}

Data Type Configuration

interface ColumnDefinition {
  type?: 'text' | 'number' | 'currency' | 'percentage' | 'date' |
         'datetime' | 'time' | 'boolean' | 'email' | 'url' | 'phone' |
         'array' | 'object';

  // Number formatting
  numberFormat?: {
    minimumFractionDigits?: number;
    maximumFractionDigits?: number;
    useGrouping?: boolean;
  };

  // Date formatting
  dateFormat?: string;

  // Currency formatting
  currencyCode?: string;
  currencySymbol?: string;
}

<ng-template #statusTemplate let-value="value" let-row="row">

      Delete
    </button>

Cell Editor Components

NgVuiGrid provides comprehensive inline editing capabilities with various built-in editor types and support for custom editors.

Text Editor

Basic text input with validation and character limits

{
  field: 'name',
  editable: true,
  cellEditor: 'text',
  cellEditorParams: {
    maxLength: 50,
    placeholder: 'Enter name'
  }
}

Number Editor

Numeric input with min/max validation and step control

{
  field: 'age',
  editable: true,
  cellEditor: 'number',
  cellEditorParams: {
    min: 0,
    max: 120,
    step: 1
  }
}

Select Editor

Dropdown selection with predefined options

{
  field: 'department',
  editable: true,
  cellEditor: 'select',
  cellEditorParams: {
    selectOptions: [
      { label: 'Engineering', value: 'eng' },
      { label: 'Sales', value: 'sales' }
    ]
  }
}

Date Editor

Date picker with range validation and formatting

{
  field: 'birthDate',
  editable: true,
  cellEditor: 'date',
  cellEditorParams: {
    format: 'yyyy-MM-dd',
    min: '1900-01-01',
    max: '2023-12-31'
  }
}

Best Practices

  • Implement server-side operations for very large datasets
  • Use trackBy functions for optimal change detection
  • Minimize template complexity

üîê Security

  • Implement proper error handling

‚ôø Accessibility
